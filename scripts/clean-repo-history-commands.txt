# Clean repo history — run from repo root, one block at a time

# --- Step 1: Orphan branch + stage all ---
git checkout --orphan temp-main
git add -A

# --- Step 2: Single commit, you as author ---
git commit -m "Initial commit" --author="Daniel Saad <dsaad68@gmail.com>"

# --- Step 3: Replace main ---
git branch -D main
git branch -m main

# --- Step 4: Force push ---
git push --force origin main

# --- Step 5 (only if Co-authored-by still in message): strip it ---
FILTER_BRANCH_SQUELCH_WARNING=1 git filter-branch -f --msg-filter 'grep -v "Co-authored-by:"' main
git push --force origin main

# --- Step 6: Cleanup backup ref ---
git update-ref -d refs/original/refs/heads/main 2>/dev/null || true
